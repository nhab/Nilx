Install and run a sample


- node --version
- npm --v
- npm install -g @angular/cli
  (newer: npm install --location=global @angular/cli)
- ng v
- install vs-code
- ng new hello-world
- cd hello-world
- ng serve
-md components
  ng generate component [name-of-component]
