Run mysql :

            mysql -u root -p 
    
Stop MySQL Server on Linux

        /etc/init.d/mysqld stop 
    
Start MySQL Server on Linux
        sudo service mysql start
    
list all databases

            SHOW DATABASES;
            or SHOW SCHEMAS;
            or SELECT schema_name FROM information_schema.schemata; 
    
Create dabase

        CREATE DATABASE movies;

    
Switch to a specific database
>
            use database_name
    
Show list of tables:

            SHOW TABLES;
    
Creating a table

        CREATE TABLE movies(title VARCHAR(50) NOT NULL,genre VARCHAR(30) NOT NULL,
        director VARCHAR(60) NOT NULL,release_year INT NOT NULL,PRIMARY KEY(title));
    
List all columns of table1

            DESCRIBE table1;

    
Show all users

        SELECT user FROM mysql.user;

    
Show current user

        SELECT user();
    
Creating a user

            CREATE USER [IF NOT EXISTS] account_name IDENTIFIED BY 'password';
    
Change user`s password

        UPDATE user 
                SET password = PASSWORD('dolphin')
            WHERE user = 'dbadmin' AND 
              host = 'localhost';

        or:
            SET PASSWORD FOR 'dbadmin'@'localhost' = PASSWORD('bigshark');
